Sample Data Transform (Code Highlight)


-- Example: Compute quarterly net interest income
SELECT
  quarter,
  SUM(interest_income) - SUM(interest_expense) AS net_interest_income
FROM finance_ledger
GROUP BY quarter
ORDER BY quarter;
